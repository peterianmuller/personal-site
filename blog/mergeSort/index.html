MergeSort derives from the truth that an array of one or 0 elements represents a sorted array. Merge Sort is an algorithm that uses a helper function merge that merges two sorted arrays. We run this merge on the left hand side of the array then run mergeSort on the right side of the array. We then return one final merge at the end of the mergseSort function. The final call will eventually be the merge on the sorted left hand side of the array and the sorted right hand side

merge = (sortedArr1, sortedArr2) => {
	let mergedSortedArray = [],
		i = 0,
		j = 0;
	// while sortedArr1 and sortedArr2 both have a length
	while (i < sortedArr1.length && j < sortedArr2.length) {
		// check if sortedArr1 at index i is greater than sortedArr2 at index j
		if (sortedArr1[i] < sortedArr2[j]) {
			// if true push element in sortedArr1 at index i to mergedSortedArray
			mergedSortedArray.push(sortedArr1[i]);
			i++;
		} else {
			// else push element in sortedArr2 at index j to mergedSortedArray
			mergedSortedArray.push(sortedArr2[j]);
			j++;
		}
	}
	// iterate over remainder of sortedArr1 and push each element into mergedSortedArray
	while (i < sortedArr1.length) {
		mergedSortedArray.push(sortedArr1[i]);
		i++;
	}
	// iterate over remainder of sortedArr2 and push each element into mergedSortedArray
	while (j < sortedArr2.length) {
		mergedSortedArray.push(sortedArr2[j]);
		j++;
	}
	return mergedSortedArray;
};

mergeSort = arr => {
	// if arr is length of 1
	if (arr.length === 1) {
		// return arr
		return arr;
	}
	// invoke mergeSort on left half of arr
	let left = mergeSort(arr.slice(0, Math.floor(arr.length / 2)));
	// invoke mergeSort on right half of arr
	let right = mergeSort(arr.slice(Math.floor(arr.length / 2)));
	// return merge of left and right
	return merge(left, right);
};

Time Complexity is O(n log n). We have n comparisons for log n subarrays. It takes us log base 2 of n to get to arrays of length one. For an array of length 4 is takes 2 decompositions. For an array of length 8 it takes 3 decompositions to get down to arrays of length 1.
