This problem comes from Giacomo Sorbi.

The goal is to here is to have a function take in an integer and have that integer return the next biggest number with those same digits.

Some examples listed on the codewars profile for this problem:

nextBigger(12)==21
nextBigger(513)==531
nextBigger(2017)==2071

I'd like to walk through a dope solution that I found. Shout outs to groxxda and Sagobunny!


function nextBigger(n){

  //create an array of single string characters for each digit in the input integer.
  var chars = n.toString().split('');

 
  
  //set a variable i to be the one less than the length of the array. Note that this number will also correspond to the terminal index of the array.
  var i = chars.length-1;
 

  
  while(i > 0) {

  //comparing the last element in the chars array to the element to its left. If the last digit is greater than the previous digit, than break out. This means that a bigger integer can be made. 
  
  if (chars[i]>chars[i-1]) break;
    
  //If the terminal element of the chars array is not larger than the digit that comes before it, than decrement one from i. This will now compare the penultimate element to the element that comes before it.
  
  i--;
  
  }
  
  //if we have compared all elements and the elements to the left of each element and the element that comes to the right is always smaller, than i will be 0 and we will break out of the while loop.
  if (i == 0) return -1;
  

  // if an element to the right is larger than its neighbor to the left, than i represents the index of the bigger element. suf will represent an array of all elements starting from the element at index i and will be a new array that is removed from chars and saved to suf.

  var suf = chars.splice(i).sort();

  //t is the terminal element of the remaining elements of char.
  var t = chars[chars.length-1];
  
  //iterate for all elements in suf
  for (i = 0; i < suf.length; ++i) {
    
  //check if suf has a digit that is bigger than the last element in what remains in char. If it does, i represent the index where the bigger integer exists. If there is no digit in suf that is larger than the terminal element of chars than i will be the terminal element of suf.
  
  if (suf[i] > t) break;
  
  }
  
  //Set the last element of chars to either 1) the larger element from suf or 2) the terminal element of suf.
  chars[chars.length-1] = suf[i]
  
  //Set either 1) the larger element of suf or 2) the last element of suf to t. t is the terminal element of chars.
  suf[i] = t;
  

  //concat chars array with the suf array
  var res = chars.concat(suf);
  
  //turn res back into an integer by first turning into a string with .join and then parseInt to convert the string to an integer.
  var num = parseInt(res.join(''));
 
  return num;
  
};

nextBigger(12); //21
nextBigger(513); //531
nextBigger(2017); //2071
